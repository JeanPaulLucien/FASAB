    if #varNum = 1

        else if #varNum = 3
                set #result Prompt(!"Enter your heroe level (1...13):" 1 2 3 4 5 6 7 8 9) // 1-13
            else if #varNum = 4
                set #result Prompt(!"Do you wish just to be on the planet?" "Yes" "No, I wish battle") // 1 или 2
            end_if
        end_if
    end_if
    if #result = 0
        goto retryInputUserVar
    end_if
    set $result #result
end_proc
